  17. const string init = @ " import clr clr.AddReference ('System.Windows.Browser') from System.Windows.Browser import *";

  20. initSource.Execute (scope);
  21. var script = engine.CreateScriptSourceFromString (source, SourceCodeKind.Statements);
  22. script.Execute (scope);
  24. return scope;
