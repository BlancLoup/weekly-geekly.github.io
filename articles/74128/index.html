Configuring Vim for Python

Today I want to tell you how I set up everyone‚Äôs favorite Vim editor (who squeaks in one mode and spoils everything in the other) for writing Python code. The article is designed for beginners vimmers.
I got this picture:


.vimrc


First, you need to set some parameters in ~ / .vimrc . I cite the entire file, maybe someone will be useful.

" VI:
set nocompatible

" - :
filetype on
filetype plugin on

" Python,
set tabstop=4
set shiftwidth=4
set smarttab
set expandtab "
set softtabstop=4 "4
"
set autoindent
"
let python_highlight_all = 1
" 256 , ?
" , gnome-terminal
set t_Co=256

" omnicomletion Python ( js, html css)
autocmd FileType python set omnifunc=pythoncomplete#Complete
autocmd FileType javascript set omnifunc=javascriptcomplete#CompleteJS
autocmd FileType html set omnifunc=htmlcomplete#CompleteTags
autocmd FileType css set omnifunc=csscomplete#CompleteCSS

"
function InsertTabWrapper()
let col = col('.') - 1
if !col || getline('.')[col - 1] !~ '\k'
return "\"
else
return "\<cp>"
endif
endfunction
imap <cr>=InsertTabWrapper()"
set complete=""
set complete+=.
set complete+=k
set complete+=b
set complete+=t

" ( .py )
autocmd BufWritePre *.py normal m`:%s/\s\+$//e ``
" .py
autocmd BufRead *.py set smartindent cinwords=if,elif,else,for,while,try,except,finally,def,class

"""" ,
"""" ,

" SnippletsEmu(. ) ctrl-j
" tab ( )
let g:snippetsEmu_key = "<Cj>"

" / Ctrl+C/Ctrl+V
vmap <CC> "+yi
imap <CV> "+gPi

colorscheme wombat256 "
syntax on "
set nu "
set mousehide "
set mouse=a "
set termencoding=utf-8 "
set novisualbell "
set t_vb= " ! ( ' ', , )
" backspace
set backspace=indent,eol,start whichwrap+=<,>,[,]
"
set showtabline=0
", :
set foldcolumn=1

" ,
set wrap
set linebreak

" .swp ~ ()
set nobackup
set noswapfile
set encoding=utf-8 "
set fileencodings=utf8,cp1251 " , unicode ,
" cp1251
